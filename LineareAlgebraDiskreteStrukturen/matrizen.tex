	\section*{Schreibweisen von Matrizen}
		\begin{align*}
			&A \in \mathbb{K}^{n,p},
			B \in \mathbb{K}^{p,m},
			C \in \mathbb{K}^{n,m}
			\\
			&A = (a_{ij})_{i=1,...,m ~~ j=1,...,p}
			\\
			&(A)_{ij} = a_{ij} = A_{(ij)}
			\\
			&C := A \cdot B
				= (c_{jk})_{j=1,...,n ~~ k=1,...,m}
				= \left( \sum\limits_{i=1}^{p} a_{ji} b_{i,k} \right)_{j=1,...,n ~~ k=1,...,m}
			\\
			&&c_{jk}
				= \sum\limits_{i=1}^{p} a_{ji} b_{i,k}
		\end{align*}
